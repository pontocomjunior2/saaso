1. Adicionar 2 actions ao `useKanbanStore`:

```typescript
createStage: async (pipelineId: string, name: string) => {
  try {
    await api.post('/stages', { pipelineId, name });
    await get().fetchPipeline(pipelineId);
  } catch (error: any) {
    set({ error: error?.response?.data?.message || 'Erro ao criar etapa' });
    throw error;
  }
},

loadTemplate: async (pipelineId: string, templateId: string) => {
  try {
    // Deleta o pipeline vazio e cria um novo a partir do template
    // Alternativa: criar from-template e redirecionar
    const response = await api.post('/pipelines/from-template', { templateId });
    // Redirecionar para o novo pipeline sera feito pelo componente
    return response.data;
  } catch (error: any) {
    set({ error: error?.response?.data?.message || 'Erro ao carregar template' });
    throw error;
  }
},
```

Atualizar a interface `KanbanState` para incluir estes 2 metodos.
