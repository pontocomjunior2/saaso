```typescript
import { IsString, IsNotEmpty, IsEnum, IsOptional } from 'class-validator';
import { CampaignChannel } from '@prisma/client';

export class CreateStageMessageTemplateDto {
  @IsString() @IsNotEmpty() stageId: string;
  @IsString() @IsNotEmpty() name: string;
  @IsEnum(CampaignChannel) channel: CampaignChannel;
  @IsString() @IsOptional() subject?: string;
  @IsString() @IsNotEmpty() body: string;
}
```

3. **Service** (`stage-message-template.service.ts`):
- `findAllByStage(tenantId, stageId)`: retorna templates da etapa, validando que o stage pertence ao tenant (join via stage.pipeline.tenantId). Ordenar por `createdAt asc`.
- `findOne(tenantId, id)`: busca template por ID, valida tenant.
- `create(tenantId, dto)`: cria template. Antes, validar que o stage existe e pertence ao tenant.
- `update(tenantId, id, dto)`: atualiza campos.
- `remove(tenantId, id)`: deleta template.
