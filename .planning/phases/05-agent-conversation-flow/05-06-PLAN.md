```tsx
'use client';
import { useState } from 'react';
import api from '@/lib/api';

interface Props {
  cardId: string;
  suggestedStageId: string;
  stageName: string;      // resolved by parent from pipeline.stages lookup OR from card.latestAgentSuggestion.suggested_next_stage_name
  onMoved?: () => void;
}

export function SuggestedStageButton({ cardId, suggestedStageId, stageName, onMoved }: Props) {
  const [busy, setBusy] = useState(false);
  const [err, setErr] = useState<string | null>(null);
  const handleClick = async () => {
    setBusy(true); setErr(null);
    try {
      await api.patch(`/cards/${cardId}/move`, { targetStageId: suggestedStageId });
      onMoved?.();
    } catch {
      setErr('Não foi possível mover o card. Tente novamente.');
    } finally {
      setBusy(false);
    }
  };
  return (
    <div>
      <button
        onClick={handleClick}
        disabled={busy}
        className="inline-flex items-center gap-2 rounded-md bg-[#594ded] px-3 py-1.5 text-sm font-medium text-white hover:bg-[#4a3fd6] disabled:opacity-60"
      >
        {busy ? 'Movendo...' : `Mover para ${stageName}`}
      </button>
      {err && <p className="mt-1 text-xs text-[#fc736f]">{err}</p>}
    </div>
  );
}
```

The `stageName` is resolved at the call site by either (a) `card.latestAgentSuggestion.suggested_next_stage_name` (preferred, when rendering from Kanban) OR (b) looking up `suggested_next_stage_id` against the already-loaded `pipeline.stages` array inside CardDetailSheet (the sheet already fetches pipeline metadata). If the stage is not found, render a disabled button with label "Estágio sugerido inválido — selecione manualmente".
