<objective>
Ship the read-only unified timeline endpoint that powers the `Atendimento` tab in CardDetailSheet (UI-SPEC §Tab). Merges three sources — WhatsAppMessage (contact-scoped), CardActivity (card-scoped), AgentMessage (conversation-scoped) — into a single chronological feed without writing anything. Endpoint is fully tenant-scoped: the card's `tenantId` is verified against the JWT's tenantId before any query fans out.

Purpose: the UI needs one call to render the full conversation+audit feed; merging client-side would leak row counts and force N queries. Backend merge keeps one network round-trip and one tenant check.
Output: `GET /cards/:id/timeline?limit=100&before=ISO` returning `{ items: UnifiedTimelineEvent[], nextCursor: string | null }`.
</objective>

// The three source queries run in parallel; each scoped to tenant + card:
const [messages, activities, agentMsgs] = await Promise.all([
  this.prisma.whatsAppMessage.findMany({
    where: {
      contact: { cards: { some: { id: cardId } } },
      tenantId,
      ...(before ? { createdAt: { lt: before } } : {}),
    },
    orderBy: { createdAt: 'desc' },
    take: limit,
  }),
  this.prisma.cardActivity.findMany({
    where: {
      cardId,
      tenantId,
      ...(before ? { createdAt: { lt: before } } : {}),
    },
    orderBy: { createdAt: 'desc' },
    take: limit,
  }),
  this.prisma.agentMessage.findMany({
    where: {
      conversation: { cardId, tenantId },
      ...(before ? { createdAt: { lt: before } } : {}),
    },
    orderBy: { createdAt: 'desc' },
    take: limit,
  }),
]);

// Merge in-memory, sort desc, slice to limit:
const merged = [
  ...messages.map(m => ({ source: 'whatsapp' as const, createdAt: m.createdAt, data: m })),
  ...activities.map(a => ({ source: 'activity' as const, createdAt: a.createdAt, data: a })),
  ...agentMsgs.map(am => ({ source: 'agent' as const, createdAt: am.createdAt, data: am })),
].sort((x, y) => y.createdAt.getTime() - x.createdAt.getTime()).slice(0, limit);

const nextCursor = merged.length === limit ? merged[merged.length - 1].createdAt.toISOString() : null;
return { items: merged, nextCursor };
```

import { IsOptional, IsInt, IsISO8601, Max, Min } from 'class-validator';
import { Type } from 'class-transformer';

export class TimelineQueryDto {
  @IsOptional()
  @Type(() => Number)
  @IsInt()
  @Min(1)
  @Max(200)
  limit?: number = 100;

  @IsOptional()
  @IsISO8601()
  before?: string;
}

export type TimelineEventSource = 'whatsapp' | 'activity' | 'agent';

export interface UnifiedTimelineEvent {
  source: TimelineEventSource;
  createdAt: Date;
  data: unknown; // narrowed per source downstream
}

export interface TimelineResponse {
  items: UnifiedTimelineEvent[];
  nextCursor: string | null;
}
```

Extend `apps/api/src/card/card.service.ts` with:

```typescript
async getCardTimeline(
  cardId: string,
  tenantId: string,
  limit = 100,
  before?: Date,
): Promise<TimelineResponse> {
  const cappedLimit = Math.min(Math.max(limit, 1), 200);

  // Tenant ownership check FIRST — before any fan-out query
  const card = await this.prisma.card.findUnique({
    where: { id: cardId },
    select: { id: true, tenantId: true },
  });
  if (!card) throw new NotFoundException('Card not found');
  if (card.tenantId !== tenantId) throw new ForbiddenException('Tenant mismatch');

  const beforeClause = before ? { createdAt: { lt: before } } : {};

  const [messages, activities, agentMsgs] = await Promise.all([
    this.prisma.whatsAppMessage.findMany({
      where: {
        contact: { cards: { some: { id: cardId } } },
        tenantId,
        ...beforeClause,
      },
      orderBy: { createdAt: 'desc' },
      take: cappedLimit,
    }),
    this.prisma.cardActivity.findMany({
      where: { cardId, tenantId, ...beforeClause },
      orderBy: { createdAt: 'desc' },
      take: cappedLimit,
    }),
    this.prisma.agentMessage.findMany({
      where: { conversation: { cardId, tenantId }, ...beforeClause },
      orderBy: { createdAt: 'desc' },
      take: cappedLimit,
    }),
  ]);

  const merged: UnifiedTimelineEvent[] = [
    ...messages.map((m) => ({ source: 'whatsapp' as const, createdAt: m.createdAt, data: m })),
    ...activities.map((a) => ({ source: 'activity' as const, createdAt: a.createdAt, data: a })),
    ...agentMsgs.map((am) => ({ source: 'agent' as const, createdAt: am.createdAt, data: am })),
  ]
    .sort((x, y) => y.createdAt.getTime() - x.createdAt.getTime())
    .slice(0, cappedLimit);

  const nextCursor =
    merged.length === cappedLimit
      ? merged[merged.length - 1].createdAt.toISOString()
      : null;

  return { items: merged, nextCursor };
}
```
