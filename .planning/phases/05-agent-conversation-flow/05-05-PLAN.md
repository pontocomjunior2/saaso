Backend merge keeps one network round-trip and one tenant check. The `latestAgentSuggestion` field unblocks the Plan 05-06 frontend components without a separate endpoint.

```typescript
async getCardTimeline(
  cardId: string,
  tenantId: string,
  limit = 100,
  before?: Date,
): Promise<TimelineResponse> {
  const cappedLimit = Math.min(Math.max(limit, 1), 200);

  // Tenant ownership check FIRST — before any fan-out query
  const card = await this.prisma.card.findUnique({
    where: { id: cardId },
    select: { id: true, tenantId: true },
  });
  if (!card) throw new NotFoundException('Card not found');
  if (card.tenantId !== tenantId) throw new ForbiddenException('Tenant mismatch');

  const beforeClause = before ? { createdAt: { lt: before } } : {};

  // IMPORTANT: WhatsAppMessage and CardActivity have NO tenantId column.
  // Tenant isolation is established by the ownership check above PLUS relation scoping
  // (WhatsAppMessage via contact.cards.some.id=cardId; CardActivity via cardId).
  // Only AgentMessage uses explicit tenantId — via its conversation relation.
  const [messages, activities, agentMsgs] = await Promise.all([
    this.prisma.whatsAppMessage.findMany({
      where: {
        contact: { cards: { some: { id: cardId } } },
        ...beforeClause,
      },
      orderBy: { createdAt: 'desc' },
      take: cappedLimit,
    }),
    this.prisma.cardActivity.findMany({
      where: { cardId, ...beforeClause },
      orderBy: { createdAt: 'desc' },
      take: cappedLimit,
    }),
    this.prisma.agentMessage.findMany({
      where: { conversation: { cardId, tenantId }, ...beforeClause },
      orderBy: { createdAt: 'desc' },
      take: cappedLimit,
    }),
  ]);

  const merged: UnifiedTimelineEvent[] = [
    ...messages.map((m) => ({ source: 'whatsapp' as const, createdAt: m.createdAt, data: m })),
    ...activities.map((a) => ({ source: 'activity' as const, createdAt: a.createdAt, data: a })),
    ...agentMsgs.map((am) => ({ source: 'agent' as const, createdAt: am.createdAt, data: am })),
  ]
    .sort((x, y) => y.createdAt.getTime() - x.createdAt.getTime())
    .slice(0, cappedLimit);

  const nextCursor =
    merged.length === cappedLimit
      ? merged[merged.length - 1].createdAt.toISOString()
      : null;

  return { items: merged, nextCursor };
}
```

```typescript
private async resolveLatestAgentSuggestion(
  cardId: string,
): Promise<LatestAgentSuggestion | null> {
  // 1. Most recent AGENT_QUALIFIED activity for this card
  const qualified = await this.prisma.cardActivity.findFirst({
    where: { cardId, type: AGENT_ACTIVITY_TYPES.AGENT_QUALIFIED },
    orderBy: { createdAt: 'desc' },
    select: { id: true, createdAt: true, metadata: true },
  });
  if (!qualified) return null;

  // 2. Invalidation: a later stage-move activity consumes the suggestion.
  //    The existing card-move flow (apps/api/src/card/card.service.ts:675) writes
  //    `type: 'MOVED'` — this is the verified literal we match against.
  const laterMove = await this.prisma.cardActivity.findFirst({
    where: {
      cardId,
      type: 'MOVED',
      createdAt: { gt: qualified.createdAt },
    },
    select: { id: true },
  });
  if (laterMove) return null;

  const meta = (qualified.metadata ?? {}) as Record<string, unknown>;
  const suggestedStageId = typeof meta.suggested_next_stage_id === 'string'
    ? (meta.suggested_next_stage_id as string)
    : null;

  let suggestedStageName: string | null = null;
  if (suggestedStageId) {
    const stage = await this.prisma.stage.findUnique({
      where: { id: suggestedStageId },
      select: { name: true },
    });
    suggestedStageName = stage?.name ?? null;
  }

  return {
    mark_qualified: true,
    qualification_reason:
      typeof meta.qualification_reason === 'string' ? (meta.qualification_reason as string) : null,
    suggested_next_stage_id: suggestedStageId,
    suggested_next_stage_name: suggestedStageName,
    confirmedAt: qualified.createdAt.toISOString(),
  };
}
```
