**Step 4 — Rewrite `processInboundMessage`:**
```typescript
public async processInboundMessage(
  input:
    | { contactId: string; content: string; whatsAppMessageId: string | null }
    | AgentRetryJobPayload,
): Promise<AgentRunnerResult> {
  const normalized = normalizeInbound(input);

  // Resolve card + agent — the existing resolution logic stays the same,
  // but card.findUnique MUST include stage.pipelineId for QualificationHandler.
  // (pseudocode — keep the existing resolveCardAndAgent flow; ensure the final card read is:)
  const card = await this.prisma.card.findUnique({
    where: { id: cardId },
    include: {
      stage: { select: { id: true, pipelineId: true } },
      // plus other fields the current runner already reads (title, tenantId, etc.)
    },
  });
  if (!card) return { status: 'no_agent' };

  // Resolve/create AgentConversation; capture updatedAt as lock token
  const conversation = await this.prisma.agentConversation.findFirst({
    where: { cardId, agentId: agent.id, status: { not: 'CLOSED' } },
    select: { id: true, summary: true, updatedAt: true, status: true, tenantId: true, contactId: true },
  });
  const capturedAt = conversation.updatedAt;

  // G8 opt-out — BEFORE persisting USER row
  if (OPT_OUT_PATTERN.test(normalized.content)) {
    await this.prisma.cardActivity.create({
      data: { cardId, type: AGENT_ACTIVITY_TYPES.LEAD_OPT_OUT, content: `Lead solicitou parar de receber mensagens: "${normalized.content.slice(0, 120)}"` },
    });
    await this.prisma.agentConversation.updateMany({
      where: { id: conversation.id, updatedAt: capturedAt },
      data: { status: AgentConversationStatus.CLOSED, lastMessageAt: new Date() },
    });
    this.notifications.emit(tenantId, { type: 'LEAD_OPT_OUT', cardId, cardTitle: card.title ?? 'Lead', at: new Date().toISOString() });
    try { await this.whatsappService.sendMessage(normalized.contactId, OPT_OUT_CONFIRMATION); } catch (err) { this.logger.warn(`Opt-out confirmation failed: ${err}`); }
    return { status: 'opted_out', conversationId: conversation.id, agentId: agent.id };
  }

  // Persist USER AgentMessage (metadata:null — D-12 reserves metadata for AGENT turns)
  await this.prisma.agentMessage.create({
    data: {
      conversationId: conversation.id,
      role: AgentMessageRole.USER,
      content: normalized.content,
      whatsAppMessageId: normalized.whatsAppMessageId,
    },
  });

  const profile = normalizeAgentPromptProfile(agent.profile);
  const compiledPrompt = buildAgentCompiledPrompt(agent, profile);
  const history = await this.historyLoader.load({
    conversationId: conversation.id,
    windowSize: profile?.historyWindow ?? 20,
  });

  let generated;
  try {
    generated = await this.replyGenerator.generate({
      compiledPrompt,
      userMessage: normalized.content,
      history,
      summary: conversation.summary,
      profile,
    });
  } catch (err) {
    if (err instanceof AgentProviderError) {
      await this.prisma.cardActivity.create({
        data: {
          cardId,
          type: AGENT_ACTIVITY_TYPES.AGENT_ERROR,
          content: 'Erro transitório ao chamar o modelo; re-tentando em background.',
          metadata: { reason: err.reason, raw: err.raw } as Prisma.InputJsonValue,
        },
      });
      await this.retryQueue.enqueue({
        conversationId: conversation.id,
        cardId,
        tenantId,
        agentId: agent.id,
        contactId: normalized.contactId,
        inboundContent: normalized.content,       // preserves the retry-payload field name
        whatsAppMessageId: normalized.whatsAppMessageId,
        enqueuedAt: new Date().toISOString(),
      });
      return { status: 'retrying', conversationId: conversation.id, agentId: agent.id };
    }
    throw err;
  }

  if (generated.fallback && generated.fallbackReason === 'parse') {
    await this.prisma.cardActivity.create({
      data: {
        cardId,
        type: AGENT_ACTIVITY_TYPES.AGENT_PARSE_FALLBACK,
        content: 'Modelo retornou JSON inválido; usando fallback.',
        metadata: { raw_output: generated.rawOutput } as Prisma.InputJsonValue,
      },
    });
    // continue — D-18
  }

  if (generated.fallback && generated.fallbackReason === 'refusal') {
    await this.prisma.cardActivity.create({
      data: {
        cardId,
        type: AGENT_ACTIVITY_TYPES.AGENT_HELD,
        content: 'Modelo recusou a resposta.',
        metadata: { reason: 'model_refusal', raw_refusal: generated.rawOutput } as Prisma.InputJsonValue,
      },
    });
    this.notifications.emit(tenantId, { type: 'AGENT_REFUSAL_REVIEW', cardId, cardTitle: card.title ?? 'Lead', at: new Date().toISOString() });
    await this.optimisticUpdate(conversation.id, capturedAt, { lastMessageAt: new Date() });
    return { status: 'held', conversationId: conversation.id, agentId: agent.id, reason: 'model_refusal' };
  }

  // CASCADE — handoff first (HARD STOP)
  if (generated.reply.request_handoff) {
    const result = await this.handoffHandler.apply({
      reply: generated.reply,
      conversation,
      card: { id: cardId },
      agentName: agent.name,
      rawOutput: generated.rawOutput,
    });
    return { status: 'handoff_required', conversationId: result.conversationId, agentId: agent.id };
  }

  // Qualification — NON-terminal. Pass pipelineId resolved via card.stage.pipelineId.
  if (generated.reply.mark_qualified) {
    await this.qualificationHandler.apply({
      reply: generated.reply,
      card: {
        id: cardId,
        pipelineId: card.stage.pipelineId,   // <-- via relation; Card has no direct pipelineId
        title: card.title,
        tenantId,
      },
      rawOutput: generated.rawOutput,
    });
  }

  // Held (D-04, D-11) — NO AgentMessage AGENT row
  if (!generated.reply.should_respond) {
    await this.prisma.cardActivity.create({
      data: {
        cardId,
        type: AGENT_ACTIVITY_TYPES.AGENT_HELD,
        content: 'Agente segurou resposta (fragmentação).',
        metadata: {
          reason: 'prompt_driven',
          held_output: generated.reply as unknown as Prisma.InputJsonValue,
          raw_output: generated.rawOutput,
        } as Prisma.InputJsonValue,
      },
    });
    await this.optimisticUpdate(conversation.id, capturedAt, { lastMessageAt: new Date() });
    return { status: 'held', conversationId: conversation.id, agentId: agent.id, reason: 'prompt_driven' };
  }

  // Dispatch
  const dispatch = await this.outboundDispatcher.send({
    reply: generated.reply,
    conversation: { id: conversation.id, contactId: normalized.contactId },
    card: { id: cardId },
    agent: { id: agent.id, name: agent.name, knowledgeBase: agent.knowledgeBase ?? null },
    tenant: { name: agent.tenant.name },
    inboundIsDisclosureChallenge: DISCLOSURE_CHALLENGE_PATTERN.test(normalized.content),
    rawOutput: generated.rawOutput,
  });

  if (dispatch.status === 'sent') {
    const agentMsgCount = await this.prisma.agentMessage.count({
      where: { conversationId: conversation.id, role: AgentMessageRole.AGENT },
    });
    const summaryThreshold = profile?.summaryThreshold ?? 10;
    if (agentMsgCount > 0 && agentMsgCount % summaryThreshold === 0) {
      await this.summarizerQueue.enqueue(conversation.id).catch((err) =>
        this.logger.warn(`Summarizer enqueue failed: ${err}`),
      );
    }
    return { status: 'replied', conversationId: conversation.id, agentId: agent.id, whatsAppMessageId: dispatch.whatsAppMessageId };
  }

  if (dispatch.status === 'handoff_required') {
    await this.optimisticUpdate(conversation.id, capturedAt, { status: AgentConversationStatus.HANDOFF_REQUIRED, lastMessageAt: new Date() });
    return { status: 'handoff_required', conversationId: conversation.id, agentId: agent.id };
  }

  await this.optimisticUpdate(conversation.id, capturedAt, { lastMessageAt: new Date() });
  return { status: 'held', conversationId: conversation.id, agentId: agent.id, reason: dispatch.reason };
}
```

```typescript
private async optimisticUpdate(conversationId: string, capturedAt: Date, data: Prisma.AgentConversationUpdateInput): Promise<void> {
  const res = await this.prisma.agentConversation.updateMany({
    where: { id: conversationId, updatedAt: capturedAt },
    data,
  });
  if (res.count === 0) {
    this.logger.warn(`Optimistic lock miss on conversation ${conversationId}; state advanced concurrently.`);
  }
}
```
