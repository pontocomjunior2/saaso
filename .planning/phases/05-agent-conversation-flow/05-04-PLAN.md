  input:
    | { contactId: string; content: string; whatsAppMessageId: string | null }
    | AgentRetryJobPayload,
): Promise<AgentRunnerResult> {

  // Resolve/create AgentConversation with the lock-token read:

    });
    await this.prisma.agentConversation.updateMany({
      where: { id: conversation.id, updatedAt: capturedAt },
      data: { status: AgentConversationStatus.CLOSED, lastMessageAt: new Date() },
    });

  const compiledPrompt = buildAgentCompiledPrompt(agent, profile);

      compiledPrompt, userMessage: input.content, history, summary: conversation.summary, profile,
    });
  } catch (err) {

      await this.prisma.cardActivity.create({
        data: { cardId, type: AGENT_ACTIVITY_TYPES.AGENT_ERROR, content: 'Erro transitĂłrio ao chamar o modelo; re-tentando em background.', metadata: { reason: err.reason, raw: err.raw } as Prisma.InputJsonValue },

    }
    throw err;
  }

    await this.prisma.cardActivity.create({
      data: { cardId, type: AGENT_ACTIVITY_TYPES.AGENT_PARSE_FALLBACK, content: 'Modelo retornou JSON invĂˇlido; usando fallback.', metadata: { raw_output: generated.rawOutput } as Prisma.InputJsonValue },
    });
    // Continue â€” fallback still dispatches (D-18)
  }

  if (generated.fallback && generated.fallbackReason === 'refusal') {
    await this.prisma.cardActivity.create({
      data: { cardId, type: AGENT_ACTIVITY_TYPES.AGENT_HELD, content: 'Modelo recusou a resposta.', metadata: { reason: 'model_refusal', raw_refusal: generated.rawOutput } as Prisma.InputJsonValue },
    });

    return { status: 'held', conversationId: conversation.id, agentId: agent.id, reason: 'model_refusal' };

    await this.qualificationHandler.apply({ reply: generated.reply, card: { id: cardId, pipelineId: card.pipelineId, title: card.title, tenantId }, rawOutput: generated.rawOutput });

    });
    await this.optimisticUpdate(conversation.id, capturedAt, { lastMessageAt: new Date() });
    return { status: 'held', conversationId: conversation.id, agentId: agent.id, reason: 'prompt_driven' };
  }

  // Dispatch
  const dispatch = await this.outboundDispatcher.send({
    reply: generated.reply,
    conversation: { id: conversation.id, contactId: input.contactId },
    card: { id: cardId },
    agent: { id: agent.id, name: agent.name, knowledgeBase: agent.knowledgeBase ?? null },
    tenant: { name: agent.tenant.name },
    inboundIsDisclosureChallenge: DISCLOSURE_CHALLENGE_PATTERN.test(input.content),
    rawOutput: generated.rawOutput,
  });

    const agentMsgCount = await this.prisma.agentMessage.count({ where: { conversationId: conversation.id, role: AgentMessageRole.AGENT } });
    const summaryThreshold = profile?.summaryThreshold ?? 10;
    if (agentMsgCount > 0 && agentMsgCount % summaryThreshold === 0) {
      await this.summarizerQueue.enqueue(conversation.id).catch((err) => this.logger.warn(`Summarizer enqueue failed: ${err}`));
    }
    return { status: 'replied', conversationId: conversation.id, agentId: agent.id, whatsAppMessageId: dispatch.whatsAppMessageId };
  }

  if (dispatch.status === 'handoff_required') {
    // G7 forced handoff â€” dispatcher already wrote the deflection activity; mark conversation
    await this.optimisticUpdate(conversation.id, capturedAt, { status: AgentConversationStatus.HANDOFF_REQUIRED, lastMessageAt: new Date() });
    return { status: 'handoff_required', conversationId: conversation.id, agentId: agent.id };
  }

  // dispatch.status === 'held' (G5 throttle or empty reply)
  await this.optimisticUpdate(conversation.id, capturedAt, { lastMessageAt: new Date() });
  return { status: 'held', conversationId: conversation.id, agentId: agent.id, reason: dispatch.reason };
}
