**Create `apps/web/src/stores/useWhatsAppAccountStore.ts`:**

```typescript
import { create } from 'zustand';
import { api } from '@/lib/api';

export interface WhatsAppAccount {
  id: string;
  tenantId: string;
  phoneNumber: string;
  status: 'CONNECTED' | 'DISCONNECTED' | 'QR_READY' | 'ERROR';
  provider: 'meta_cloud' | 'evolution';
  instanceName: string | null;
  createdAt: string;
  updatedAt: string;
}

interface WhatsAppAccountState {
  account: WhatsAppAccount | null;
  qrCode: string | null;
  connectionState: string | null;
  loading: boolean;
  error: string | null;

  fetchAccount: () => Promise<void>;
  createAccount: (provider: 'meta_cloud' | 'evolution', phoneNumber?: string) => Promise<void>;
  createEvolutionInstance: (instanceName: string) => Promise<void>;
  fetchQrCode: (instanceName: string) => Promise<void>;
  fetchConnectionState: (instanceName: string) => Promise<string>;
  disconnectAccount: (id: string) => Promise<void>;
  deleteAccount: (id: string) => Promise<void>;
  switchProvider: (provider: 'meta_cloud' | 'evolution') => Promise<void>;
}

export const useWhatsAppAccountStore = create<WhatsAppAccountState>((set, get) => ({
  account: null,
  qrCode: null,
  connectionState: null,
  loading: false,
  error: null,

  fetchAccount: async () => {
    set({ loading: true, error: null });
    try {
      const res = await api.get('/whatsapp/accounts');
      const account = res.data[0] ?? null;
      set({ account, loading: false });
    } catch (err: any) {
      set({ error: err?.response?.data?.message ?? 'Erro ao carregar conta WhatsApp', loading: false });
    }
  },

  createAccount: async (provider, phoneNumber) => {
    set({ loading: true, error: null });
    try {
      const res = await api.post('/whatsapp/accounts', {
        provider,
        phoneNumber: phoneNumber ?? '',
      });
      set({ account: res.data, loading: false });
    } catch (err: any) {
      set({ error: err?.response?.data?.message ?? 'Erro ao criar conta WhatsApp', loading: false });
    }
  },

  createEvolutionInstance: async (instanceName: string) => {
    set({ loading: true, error: null });
    try {
      const res = await api.post('/whatsapp/evolution/instance', { instanceName });
      set({ connectionState: 'created', loading: false });
      return res.data;
    } catch (err: any) {
      set({ error: err?.response?.data?.message ?? 'Erro ao criar instancia', loading: false });
    }
  },

  fetchQrCode: async (instanceName: string) => {
    try {
      const res = await api.get(`/whatsapp/evolution/instance/${instanceName}/qr`);
      set({ qrCode: res.data.qr });
    } catch (err: any) {
      set({ error: err?.response?.data?.message ?? 'Erro ao obter QR Code' });
    }
  },

  fetchConnectionState: async (instanceName: string) => {
    try {
      const res = await api.get(`/whatsapp/evolution/instance/${instanceName}/connection-state`);
      const state = res.data.state ?? 'unknown';
      set({ connectionState: state });
      return state;
    } catch (err: any) {
      set({ error: err?.response?.data?.message ?? 'Erro ao verificar conexao' });
      return 'error';
    }
  },

  disconnectAccount: async (id: string) => {
    try {
      await api.post(`/whatsapp/accounts/${id}/disconnect`);
      set((s) => ({
        account: s.account ? { ...s.account, status: 'DISCONNECTED' as const } : null,
        qrCode: null,
        connectionState: 'disconnected',
      }));
    } catch (err: any) {
      set({ error: err?.response?.data?.message ?? 'Erro ao desconectar' });
    }
  },

  deleteAccount: async (id: string) => {
    try {
      await api.delete(`/whatsapp/accounts/${id}`);
      set({ account: null, qrCode: null, connectionState: null });
    } catch (err: any) {
      set({ error: err?.response?.data?.message ?? 'Erro ao remover conta' });
    }
  },

  switchProvider: async (provider) => {
    if (!get().account) return;
    try {
      const res = await api.patch(`/whatsapp/accounts/${get().account!.id}`, { provider });
      set({ account: res.data, qrCode: null, connectionState: null });
    } catch (err: any) {
      set({ error: err?.response?.data?.message ?? 'Erro ao trocar provider' });
    }
  },
}));
```

**Create `apps/web/src/components/configuracoes/WhatsAppSettingsSection.tsx`:**

```tsx
'use client';

import { useState, useEffect, useCallback, useRef } from 'react';
import { Loader2, QrCode, Smartphone, CheckCircle2, AlertCircle, RefreshCw, Wifi, WifiOff } from 'lucide-react';
import { useWhatsAppAccountStore, WhatsAppAccount } from '@/stores/useWhatsAppAccountStore';

const STATUS_CONFIG: Record<string, { label: string; color: string; bg: string; icon: React.ReactNode }> = {
  CONNECTED: { label: 'Conectado', color: 'text-emerald-600', bg: 'bg-emerald-50', icon: <CheckCircle2 className="h-4 w-4" /> },
  DISCONNECTED: { label: 'Desconectado', color: 'text-gray-500', bg: 'bg-gray-100', icon: <WifiOff className="h-4 w-4" /> },
  QR_READY: { label: 'Aguardando QR Code', color: 'text-amber-600', bg: 'bg-amber-50', icon: <QrCode className="h-4 w-4" /> },
  ERROR: { label: 'Erro na conexao', color: 'text-rose-600', bg: 'bg-rose-50', icon: <AlertCircle className="h-4 w-4" /> },
};

const CONNECTION_STATE_LABELS: Record<string, { label: string; color: string; icon: React.ReactNode }> = {
  qrcode: { label: 'Escaneie o QR Code abaixo', color: 'text-amber-600', icon: <QrCode className="h-5 w-5" /> },
  connecting: { label: 'Conectando...', color: 'text-blue-600', icon: <Loader2 className="h-5 w-5 animate-spin" /> },
  connected: { label: 'WhatsApp conectado!', color: 'text-emerald-600', icon: <CheckCircle2 className="h-5 w-5" /> },
  disconnected: { label: 'Desconectado', color: 'text-gray-500', icon: <WifiOff className="h-5 w-5" /> },
};

export function WhatsAppSettingsSection() {
  const { account, qrCode, connectionState, loading, error, fetchAccount, createEvolutionInstance, fetchQrCode, fetchConnectionState, disconnectAccount, deleteAccount, switchProvider } = useWhatsAppAccountStore();

  const [selectedProvider, setSelectedProvider] = useState<'meta_cloud' | 'evolution'>(account?.provider ?? 'evolution');
  const [instanceName, setInstanceName] = useState('');
  const [isPolling, setIsPolling] = useState(false);
  const pollRef = useRef<ReturnType<typeof setInterval> | null>(null);

  // Load account on mount
  useEffect(() => {
    fetchAccount();
  }, []);

  // Update selectedProvider when account loads
  useEffect(() => {
    if (account) setSelectedProvider(account.provider);
  }, [account?.provider]);

  // Poll connection state during QR flow
  const startPolling = useCallback(async (name: string) => {
    setIsPolling(true);
    pollRef.current = setInterval(async () => {
      const state = await fetchConnectionState(name);
      if (state === 'connected') {
        stopPolling();
        fetchAccount(); // refresh account status
      }
      if (state === 'qrcode') {
        fetchQrCode(name);
      }
    }, 3000);
  }, []);

  const stopPolling = useCallback(() => {
    if (pollRef.current) {
      clearInterval(pollRef.current);
      pollRef.current = null;
    }
    setIsPolling(false);
  }, []);

  // Cleanup on unmount
  useEffect(() => stopPolling, []);

  const handleCreateInstance = async () => {
    if (!instanceName.trim()) return;
    await createEvolutionInstance(instanceName.trim());
    startPolling(instanceName.trim());
    // Also update the account with instanceName
    // (backend should do this automatically on instance creation)
  };

  const handleRefreshQr = async () => {
    if (account?.instanceName) {
      await fetchQrCode(account.instanceName);
    }
  };

  // -- Render --
  return (
    <div className="rounded-[30px] border border-white/10 bg-[rgba(8,18,34,0.82)] p-6">
      <h2 className="text-white text-lg font-bold">WhatsApp</h2>
      <p className="text-slate-400 text-sm mt-1">Configure o canal de WhatsApp para envio e recebimento de mensagens.</p>

      {/* Provider Selection */}
      <div className="mt-6 flex gap-3">
        <button
          onClick={() => switchProvider('evolution')}
          className={`flex-1 rounded-xl border px-4 py-3 text-sm font-bold transition ${
            selectedProvider === 'evolution'
              ? 'border-[#594ded] bg-[#594ded]/10 text-[#594ded]'
              : 'border-white/10 bg-white/[0.04] text-slate-400 hover:text-white'
          }`}
        >
          Evolution API
        </button>
        <button
          onClick={() => switchProvider('meta_cloud')}
          className={`flex-1 rounded-xl border px-4 py-3 text-sm font-bold transition ${
            selectedProvider === 'meta_cloud'
              ? 'border-[#594ded] bg-[#594ded]/10 text-[#594ded]'
              : 'border-white/10 bg-white/[0.04] text-slate-400 hover:text-white'
          }`}
        >
          Meta Cloud API
        </button>
      </div>

      {/* Error */}
      {error && (
        <div className="mt-4 rounded-xl bg-rose-500/10 border border-rose-500/20 px-4 py-3 text-sm text-rose-400">
          {error}
        </div>
      )}

      {/* Evolution API Flow */}
      {selectedProvider === 'evolution' && (
        <div className="mt-6 space-y-4">
          {/* Account status */}
          {account && (
            <div className="rounded-[24px] border border-white/10 bg-white/[0.04] p-4">
              <div className="flex items-center gap-3">
                {STATUS_CONFIG[account.status]?.icon ?? <WifiOff className="h-4 w-4 text-gray-400" />}
                <span className={`text-sm font-medium ${STATUS_CONFIG[account.status]?.color ?? 'text-gray-400'}`}>
                  {STATUS_CONFIG[account.status]?.label ?? account.status}
                </span>
                {account.instanceName && (
                  <span className="ml-auto text-xs font-mono text-slate-500">{account.instanceName}</span>
                )}
              </div>
              <div className="mt-3 flex gap-2">
                {account.status === 'CONNECTED' && (
                  <button
                    onClick={() => disconnectAccount(account.id)}
                    className="rounded-xl border border-white/10 bg-white/[0.06] text-white px-3 py-2 text-xs font-bold hover:bg-white/[0.10]"
                  >
                    Desconectar
                  </button>
                )}
                <button
                  onClick={() => deleteAccount(account.id)}
                  className="rounded-xl border border-rose-500/20 bg-rose-500/10 text-rose-400 px-3 py-2 text-xs font-bold hover:bg-rose-500/20"
                >
                  Remover conta
                </button>
              </div>
            </div>
          )}

          {/* Instance creation */}
          {(!account || account.status === 'DISCONNECTED' || account.status === 'ERROR') && (
            <div>
              <label className="text-sm text-slate-400">Nome da instancia</label>
              <input
                type="text"
                value={instanceName}
                onChange={(e) => setInstanceName(e.target.value)}
                placeholder="ex: minha-empresa"
                className="mt-1 w-full rounded-xl border border-white/10 bg-white/[0.06] px-3 py-2 text-sm text-white placeholder:text-slate-500 focus:border-[#594ded] focus:outline-none"
              />
              <button
                onClick={handleCreateInstance}
                disabled={loading || !instanceName.trim()}
                className="mt-3 rounded-xl bg-[#594ded] text-white px-4 py-2 text-sm font-bold disabled:opacity-50 hover:bg-[#4b3dd4]"
              >
                {loading ? <Loader2 className="h-4 w-4 animate-spin inline" /> : 'Criar instancia'}
              </button>
            </div>
          )}

          {/* QR Code display */}
          {(connectionState === 'qrcode' || qrCode) && (
            <div className="rounded-[24px] border border-white/10 bg-white/[0.04] p-6 text-center">
              <div className="flex items-center justify-center gap-2 text-amber-400 mb-4">
                <QrCode className="h-5 w-5" />
                <span className="text-sm font-bold">Escaneie o QR Code com seu WhatsApp</span>
              </div>
              {qrCode ? (
                <div className="inline-block rounded-2xl bg-white p-4">
                  <img
                    src={qrCode.startsWith('data:') ? qrCode : `data:image/png;base64,${qrCode}`}
                    alt="QR Code WhatsApp"
                    className="w-64 h-64"
                  />
                </div>
              ) : (
                <Loader2 className="h-8 w-8 animate-spin mx-auto text-slate-400" />
              )}
              <button
                onClick={handleRefreshQr}
                className="mt-4 inline-flex items-center gap-2 rounded-xl border border-white/10 bg-white/[0.06] text-white px-3 py-2 text-xs font-bold"
              >
                <RefreshCw className="h-3 w-3" />
                Atualizar QR Code
              </button>
              {isPolling && (
                <p className="mt-3 text-xs text-slate-500 flex items-center justify-center gap-1">
                  <Loader2 className="h-3 w-3 animate-spin" />
                  Aguardando conexao...
                </p>
              )}
            </div>
          )}

          {/* Connection state labels */}
          {connectionState && CONNECTION_STATE_LABELS[connectionState] && (
            <div className={`flex items-center gap-2 text-sm ${CONNECTION_STATE_LABELS[connectionState].color}`}>
              {CONNECTION_STATE_LABELS[connectionState].icon}
              <span>{CONNECTION_STATE_LABELS[connectionState].label}</span>
            </div>
          )}
        </div>
      )}

      {/* Meta Cloud API (read-only placeholder) */}
      {selectedProvider === 'meta_cloud' && (
        <div className="mt-6 rounded-[24px] border border-white/10 bg-white/[0.04] p-4">
          <p className="text-slate-400 text-sm">
            Meta Cloud API configurada via variaveis de ambiente. Para alterar, atualize <code className="text-xs bg-white/[0.06] px-1 py-0.5 rounded">META_PHONE_NUMBER_ID</code> e <code className="text-xs bg-white/[0.06] px-1 py-0.5 rounded">META_ACCESS_TOKEN</code>.
          </p>
          {account && (
            <div className="mt-3 flex items-center gap-2 text-sm text-slate-400">
              <Smartphone className="h-4 w-4" />
              <span>{account.phoneNumber}</span>
              <span className={`ml-2 inline-flex items-center gap-1 rounded-full px-2 py-0.5 text-xs ${
                account.status === 'CONNECTED' ? 'bg-emerald-50 text-emerald-700' : 'bg-gray-100 text-gray-500'
              }`}>
                {account.status === 'CONNECTED' ? 'Conectado' : 'Desconectado'}
              </span>
            </div>
          )}
        </div>
      )}
    </div>
  );
}

'use client';

import { Bot, UserCheck, Pause } from 'lucide-react';

interface AgentStatusBadgeProps {
  agentName?: string | null;
  status: 'active' | 'paused' | 'takeover' | null;
  compact?: boolean; // true for Kanban cards, false for CardDetailSheet
}

const CONFIG = {
  active: {
    label: 'Ativo',
    tooltip: 'Agente ativo — piloto automatico',
    dotColor: 'bg-emerald-500',
    textColor: 'text-emerald-700',
    bgColor: 'bg-emerald-50',
    borderColor: 'border-emerald-200',
    icon: <Bot className="h-3 w-3" />,
  },
  paused: {
    label: 'Pausado',
    tooltip: 'Regua pausada',
    dotColor: 'bg-gray-400',
    textColor: 'text-gray-600',
    bgColor: 'bg-gray-100',
    borderColor: 'border-gray-200',
    icon: <Pause className="h-3 w-3" />,
  },
  takeover: {
    label: 'Humano',
    tooltip: 'SDR assumiu a conversa',
    dotColor: 'bg-amber-500',
    textColor: 'text-amber-700',
    bgColor: 'bg-amber-50',
    borderColor: 'border-amber-200',
    icon: <UserCheck className="h-3 w-3" />,
  },
};

export function AgentStatusBadge({ agentName, status, compact = false }: AgentStatusBadgeProps) {
  if (!status) return null;

  const cfg = CONFIG[status];
  if (!cfg) return null;

  if (compact) {
    return (
      <span
        className={`inline-flex items-center gap-1 rounded-md ${cfg.bgColor} ${cfg.textColor} px-1.5 py-0.5 text-[10px] font-semibold`}
        title={cfg.tooltip}
      >
        <span className={`h-1.5 w-1.5 rounded-full ${cfg.dotColor}`} />
        {cfg.label}
      </span>
    );
  }

  // Full size (for CardDetailSheet)
  return (
    <div
      className={`inline-flex items-center gap-2 rounded-xl border ${cfg.borderColor} ${cfg.bgColor} px-3 py-2`}
      title={cfg.tooltip}
    >
      <span className={`h-2 w-2 rounded-full ${cfg.dotColor}`} />
      <span className={`text-sm font-bold ${cfg.textColor}`}>{cfg.label}</span>
      {cfg.icon}
      {agentName && <span className="text-xs text-slate-500 ml-1">{agentName}</span>}
    </div>
  );
}
```
