@Controller('forms')
export class PublicFormController {
  constructor(private readonly leadFormService: LeadFormService) {}

  @Post(':slug/submit')
  public async submitForm(
    @Param('slug') slug: string,
    @Body() dto: SubmitFormDto,
    @Req() req: Request,
  ) {
    const ip = req.ip ?? req.socket.remoteAddress ?? 'unknown';
    return this.leadFormService.submitForm(slug, dto.payload, ip);
  }
}

async submitForm(
  slug: string,
  payload: Record<string, any>,
  ip: string,
): Promise<{ success: boolean; cardId?: string; submissionId: string; contactId: string }> {
  // 1. Find form by slug
  const form = await this.prisma.leadForm.findFirst({
    where: { slug },
    include: { stage: true },
  });
  if (!form || !form.isActive) {
    throw new NotFoundException('Form not found or inactive');
  }

  // 2. Validate payload against form.fields schema
  this.validatePayload(form.fields as LeadFormField[], payload);

  // 3. Rate limit check
  await this.checkRateLimit(ip, form.tenantId);

  // 4. Create submission
  const submission = await this.prisma.leadFormSubmission.create({
    data: { formId: form.id, payload },
  });

  // 5. Extract contact data from payload via mapTo
  const contactData = this.extractContactData(form.fields as LeadFormField[], payload);

  // 6. Find or create contact
  const contact = await this.findOrCreateContact(form.tenantId, contactData);

  // 7. Create card if form has linked stage
  let cardId: string | undefined;
  if (form.stageId) {
    const card = await this.cardService.create(form.tenantId, {
      stageId: form.stageId,
      contactId: contact.id,
      title: `${contact.name ?? 'Lead'} — ${form.name}`,
    });
    cardId = card.id;

    // 8. Trigger agent proactive D0
    try {
      await this.agentRunnerService.initiateProactiveIfAssigned(
        cardId,
        form.stageId,
        form.tenantId,
      );
    } catch (err) {
      this.logger.error('[submitForm] initiateProactiveIfAssigned failed', err);
    }
  }

  return { success: true, cardId, submissionId: submission.id, contactId: contact.id };
}

private validatePayload(fields: LeadFormField[], payload: Record<string, any>): void {
  const errors: string[] = [];
  for (const field of fields) {
    const value = payload[field.key];
    if (field.required && (value === undefined || value === null || value === '')) {
      errors.push(`Field "${field.key}" is required`);
    }
    if (value && field.type === 'email' && !/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(value)) {
      errors.push(`Field "${field.key}" must be a valid email`);
    }
    if (value && field.type === 'select' && field.options && !field.options.includes(value)) {
      errors.push(`Field "${field.key}" must be one of: ${field.options.join(', ')}`);
    }
  }
  if (errors.length > 0) {
    throw new BadRequestException({ message: 'Validation failed', errors });
  }
}

private extractContactData(
  fields: LeadFormField[],
  payload: Record<string, any>,
): { name?: string; email?: string; phone?: string; companyName?: string } {
  const contact: Record<string, string> = {};
  for (const field of fields) {
    if (field.mapTo && payload[field.key]) {
      contact[field.mapTo] = payload[field.key];
    }
  }
  return contact as { name?: string; email?: string; phone?: string; companyName?: string };
}

private async findOrCreateContact(
  tenantId: string,
  data: { name?: string; email?: string; phone?: string; companyName?: string },
) {
  // Try to find by email or phone
  if (data.email || data.phone) {
    const existing = await this.prisma.contact.findFirst({
      where: {
        tenantId,
        OR: [
          ...(data.email ? [{ email: data.email }] : []),
          ...(data.phone ? [{ phone: data.phone }] : []),
        ],
      },
    });
    if (existing) return existing;
  }

  return this.prisma.contact.create({
    data: {
      tenantId,
      name: data.name ?? 'Lead sem nome',
      email: data.email,
      phone: data.phone,
      companyName: data.companyName,
    },
  });
}
```

import { Injectable, HttpException, HttpStatus } from '@nestjs/common';

interface RateLimitEntry {
  count: number;
  windowStart: Date;
}

@Injectable()
export class RateLimitService {
  private readonly limits = new Map<string, RateLimitEntry>();
  private readonly maxRequests = 5;
  private readonly windowMs = 15 * 60 * 1000; // 15 minutes

  check(ip: string, tenantId: string): void {
    const key = `${ip}:${tenantId}`;
    const now = new Date();
    const entry = this.limits.get(key);

    if (!entry || now.getTime() - entry.windowStart.getTime() > this.windowMs) {
      this.limits.set(key, { count: 1, windowStart: now });
      return;
    }

    entry.count++;
    if (entry.count > this.maxRequests) {
      throw new HttpException('Too Many Requests', HttpStatus.TOO_MANY_REQUESTS);
    }
  }

  // Exposed for testing / cleanup
  reset(): void {
    this.limits.clear();
  }
}
```

```typescript
async initiateProactiveIfAssigned(
  cardId: string,
  stageId: string,
  tenantId: string,
): Promise<void> {
  const stage = await this.prisma.stage.findFirst({
    where: { id: stageId, pipeline: { tenantId } },
    include: { agents: true },
  });
  if (!stage || stage.agents.length === 0) return;

  const agent = stage.agents[0];
  const card = await this.prisma.card.findFirst({
    where: { id: cardId, tenantId },
    include: { contact: true },
  });
  if (!card) return;

  const contact = card.contact;
  if (!contact) return;

  // Build proactive greeting prompt using existing AgentPromptBuilder
  const prompt = await this.promptBuilder.buildProactivePrompt(
    agent,
    card,
    stage.classificationCriteria,
  );
  const greeting = await this.llmService.generate(prompt);

  // Channel fallback: WhatsApp > Email > Log only
  if (contact.phone) {
    // Path 1: WhatsApp
    await this.whatsappService.sendMessage({
      tenantId,
      to: contact.phone,
      body: greeting,
    });

    await this.prisma.cardActivity.create({
      data: {
        cardId,
        type: 'AGENT_PROACTIVE_WHATSAPP',
        content: `Agente ${agent.name} enviou WhatsApp: "${greeting.substring(0, 100)}..."`,
        actorId: null,
      },
    });
  } else if (contact.email) {
    // Path 2: Email with HTML body
    const htmlBody = this.wrapGreetingInHtml(greeting, contact.name);
    await this.emailService.sendEmail({
      to: contact.email,
      subject: `Olá, ${contact.name}! Bem-vindo(a)!`,
      body: htmlBody,
      html: true,
    });

    await this.prisma.cardActivity.create({
      data: {
        cardId,
        type: 'AGENT_PROACTIVE_EMAIL',
        content: `Agente ${agent.name} enviou email para ${contact.email}: "${greeting.substring(0, 100)}..."`,
        actorId: null,
      },
    });
  } else {
    // Path 3: Log only — no channel available
    await this.prisma.cardActivity.create({
      data: {
        cardId,
        type: 'AGENT_PROACTIVE_LOGGED',
        content: `Agente ${agent.name} gerou saudação mas sem canal de envio (sem phone/email). Saudação: "${greeting.substring(0, 100)}..."`,
        actorId: null,
      },
    });
  }

  // Create AgentConversation in all cases
  await this.prisma.agentConversation.create({
    data: {
      tenantId,
      cardId,
      agentId: agent.id,
      contactId: contact.id,
      status: 'OPEN',
    },
  });
}

private wrapGreetingInHtml(greeting: string, name?: string): string {
  return `
<!DOCTYPE html>
<html>
<head><meta charset="utf-8"></head>
<body style="font-family: Arial, sans-serif; max-width: 600px; margin: 0 auto; padding: 20px;">
  <p>Olá${name ? `, ${name}` : ''}!</p>
  <p>${greeting}</p>
  <hr style="border: none; border-top: 1px solid #eee; margin: 20px 0;">
  <p style="font-size: 12px; color: #999;">Esta mensagem foi enviada automaticamente.</p>
</body>
</html>`;
}
```
