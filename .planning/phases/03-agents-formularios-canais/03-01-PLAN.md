**Modify `apps/api/src/whatsapp/whatsapp.service.ts` — convert to facade:**
```typescript
import { Injectable, Logger } from '@nestjs/common';
import { PrismaService } from '../prisma/prisma.service';
import { IWhatsAppProvider } from './providers/whatsapp-provider.interface';
import { MetaCloudProvider } from './providers/meta-cloud.provider';
import { EvolutionApiService } from './evolution.service';

@Injectable()
export class WhatsAppService {
  private readonly logger = new Logger(WhatsAppService.name);

  constructor(
    private readonly prisma: PrismaService,
    private readonly metaCloudProvider: MetaCloudProvider,
    private readonly evolutionProvider: EvolutionApiService,
  ) {}

  private async resolveProvider(tenantId: string): Promise<IWhatsAppProvider> {
    const account = await this.prisma.whatsAppAccount.findFirst({
      where: { tenantId },
    });
    if (!account) return this.metaCloudProvider; // fallback to existing behavior (backwards compatible)

    switch (account.provider) {
      case 'evolution':
        return this.evolutionProvider;
      case 'meta_cloud':
      default:
        return this.metaCloudProvider;
    }
  }

  async sendMessage(tenantId: string, to: string, body: string, html?: string): Promise<void> {
    const provider = await this.resolveProvider(tenantId);
    return provider.sendMessage(to, body, html);
  }

  async processWebhook(tenantId: string, payload: any): Promise<void> {
    const provider = await this.resolveProvider(tenantId);
    return provider.receiveWebhook(payload);
  }

  async getAccountStatus(tenantId: string): Promise<{ status: string }> {
    const provider = await this.resolveProvider(tenantId);
    return provider.getAccountStatus();
  }

  async connect(tenantId: string): Promise<void> {
    const provider = await this.resolveProvider(tenantId);
    return provider.connect();
  }

  async disconnect(tenantId: string): Promise<void> {
    const provider = await this.resolveProvider(tenantId);
    return provider.disconnect();
  }
}
```

The facade delegates `sendMessage`, `processWebhook`, `getAccountStatus`, `connect`, and `disconnect` to the resolved provider. Provider resolution uses `WhatsAppAccount.provider` field. If no account exists, defaults to MetaCloudProvider (backwards compatibility with existing tenants).

**Create `apps/api/src/whatsapp/evolution.service.ts` — EvolutionApiService implementing IWhatsAppProvider:**
```typescript
import { Injectable, Logger } from '@nestjs/common';
import { ConfigService } from '@nestjs/config';
import { IWhatsAppProvider } from './providers/whatsapp-provider.interface';
import { AgentRunnerService } from '../agent/agent-runner.service';
import { PrismaService } from '../prisma/prisma.service';

@Injectable()
export class EvolutionApiService implements IWhatsAppProvider {
  private readonly logger = new Logger(EvolutionApiService.name);
  private readonly baseUrl: string;
  private readonly globalApiKey: string;

  constructor(
    private readonly configService: ConfigService,
    private readonly agentRunner: AgentRunnerService,
    private readonly prisma: PrismaService,
  ) {
    this.baseUrl = this.configService.getOrThrow('EVOLUTION_API_URL');
    this.globalApiKey = this.configService.getOrThrow('EVOLUTION_API_KEY');
  }

  // -- Instance Management --

  async createInstance(tenantId: string, instanceName: string): Promise<any> {
    // POST {baseUrl}/instance/create
    // Body: { instanceName, webhookUrl, webhook_base64: false, reject_call: false }
    // Headers: { apikey: this.globalApiKey, 'Content-Type': 'application/json' }
    const result = await this.httpPost('/instance/create', {
      instanceName,
      webhookUrl: `${this.baseUrl}/webhook`,
      webhook_base64: false,
      reject_call: false,
    });
    // Store instanceName on WhatsAppAccount
    await this.prisma.whatsAppAccount.updateMany({
      where: { tenantId },
      data: { instanceName },
    });
    return result;
  }

  async getInstanceState(instanceName: string): Promise<string> {
    // GET {baseUrl}/instance/{instanceName}/connection-state
    const data = await this.httpGet(`/instance/${instanceName}/connection-state`);
    return data?.state ?? 'unknown';
  }

  async getQrCode(instanceName: string): Promise<string> {
    // GET {baseUrl}/instance/{instanceName}/qr-code
    const data = await this.httpGet(`/instance/${instanceName}/qr-code`);
    return data?.base64 ?? data?.qrCode ?? '';
  }

  // -- IWhatsAppProvider implementation --

  async sendMessage(to: string, body: string, html?: string): Promise<void> {
    // POST {baseUrl}/message/sendText
    // Body: { number: to, textMessage: { text: body } }
    await this.httpPost('/message/sendText', {
      number: to,
      textMessage: { text: body },
    });
    // Note: html is tracked in signature for interface parity but
    // Evolution API sendText only supports text bodies. HTML support
    // would require a different endpoint (e.g., sendWhatsAppMedia for rich content).
  }

  async receiveWebhook(payload: any): Promise<void> {
    // Parse Evolution API webhook payload
    // Extract phone number, message body, timestamp
    // Build WhatsAppInboundPayload compatible format
    // Delegate to AgentRunnerService.processInboundMessage
    const inboundPayload = {
      from: payload?.key?.remoteJid || payload?.from,
      body: payload?.message?.conversation || payload?.message?.text || payload?.message?.extendedTextMessage?.text,
      timestamp: payload?.messageTimestamp ? new Date(payload.messageTimestamp * 1000).toISOString() : new Date().toISOString(),
      tenantId: await this.resolveTenantFromInstance(payload?.instance),
    };
    await this.agentRunner.processInboundMessage(inboundPayload);
  }

  async getAccountStatus(): Promise<{ status: string }> {
    return { status: 'unknown' };
  }

  async connect(): Promise<void> {
    // Evolution API connect is implicit after QR scan — no explicit connect needed
  }

  async disconnect(): Promise<void> {
    // POST {baseUrl}/instance/logout to disconnect the instance
    await this.httpPost('/instance/logout', {});
  }

  // -- Private helpers --

  private async resolveTenantFromInstance(instanceName: string): Promise<string> {
    const account = await this.prisma.whatsAppAccount.findFirst({
      where: { instanceName },
    });
    return account?.tenantId ?? '';
  }

  private buildHeaders(): Record<string, string> {
    return {
      'Content-Type': 'application/json',
      apikey: this.globalApiKey,
    };
  }

  private async httpPost(path: string, body: any): Promise<any> {
    const response = await fetch(`${this.baseUrl}${path}`, {
      method: 'POST',
      headers: this.buildHeaders(),
      body: JSON.stringify(body),
    });
    if (!response.ok) {
      const text = await response.text();
      throw new Error(`Evolution API ${path} failed: ${response.status} ${text}`);
    }
    return response.json();
  }

  private async httpGet(path: string): Promise<any> {
    const response = await fetch(`${this.baseUrl}${path}`, {
      method: 'GET',
      headers: this.buildHeaders(),
    });
    if (!response.ok) {
      const text = await response.text();
      throw new Error(`Evolution API ${path} failed: ${response.status} ${text}`);
    }
    return response.json();
  }
}
```

```typescript
import { Controller, Post, Body, Get, Param, Logger } from '@nestjs/common';
import { EvolutionApiService } from './evolution.service';

@Controller('whatsapp/evolution')
export class EvolutionController {
  private readonly logger = new Logger(EvolutionController.name);

  constructor(private readonly evolutionService: EvolutionApiService) {}

  @Post('webhook')
  async handleWebhook(@Body() payload: any) {
    this.logger.log('Received Evolution webhook');
    await this.evolutionService.receiveWebhook(payload);
    return { status: 'received' };
  }

  @Post('instance')
  async createInstance(
    @Body() dto: { tenantId: string; instanceName: string },
  ) {
    const result = await this.evolutionService.createInstance(
      dto.tenantId,
      dto.instanceName,
    );
    return result;
  }

  @Get('instance/:name/qr')
  async getQrCode(@Param('name') name: string) {
    const qr = await this.evolutionService.getQrCode(name);
    return { qr };
  }
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Evolution API webhook (POST /whatsapp/evolution/webhook) | Unauthenticated public endpoint — must validate payload structure and resolve tenant from instance name |
| Evolution API calls outbound | Uses EVOLUTION_API_KEY from env vars — must never expose in responses or logs |
| WhatsAppAccount.apiKey field | Stored in DB — must not expose in API responses (omit from serialization) |
| Instance creation (POST /whatsapp/evolution/instance) | Authenticated — must validate tenant owns the WhatsAppAccount |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-3-01 | Information Disclosure | WhatsAppAccount.apiKey in DB | mitigate | Exclude apiKey field from all DTO serialization; use `@Exclude()` class-transformer decorator or omit in map |
| T-3-02 | Spoofing | Evolution webhook without tenant validation | mitigate | EvolutionApiService.receiveWebhook resolves tenant via instanceName lookup on WhatsAppAccount; drops webhook if tenant not found |
| T-3-03 | Tampering | EVOLUTION_API_KEY leaked in logs | mitigate | Logger must not log full API key; use masked form in any diagnostic output |
| T-3-04 | Elevation of Privilege | Instance creation without tenant check | mitigate | EvolutionController.createInstance validates requesting user's tenantId matches the target WhatsAppAccount tenantId |
| T-3-05 | Repudiation | Provider switch without audit trail | accept | WhatsAppAccount.provider change is a tenant-config operation; existing audit mechanisms (CardActivity) track outbound messages per provider |
</threat_model>
