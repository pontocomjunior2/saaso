```typescript
'use client';

import { useState, useCallback, useEffect } from 'react';
import { useSearchParams } from 'next/navigation';

type SubmissionState = 'idle' | 'loading' | 'success' | 'error';

interface FieldErrors {
  [fieldKey: string]: string;
}

// In the page component:
const [submissionState, setSubmissionState] = useState<SubmissionState>('idle');
const [fieldValues, setFieldValues] = useState<Record<string, any>>({});
const [fieldErrors, setFieldErrors] = useState<FieldErrors>({});
const [submitError, setSubmitError] = useState<string>('');

const searchParams = useSearchParams();
const isEmbed = searchParams?.get('embed') === '1';

// Client-side validation
function validateField(field: LeadFormField, value: any): string | null {
  if (field.required && (!value || (typeof value === 'string' && value.trim() === ''))) {
    return 'Este campo e obrigatorio';
  }

  if (value && field.type === 'email') {
    const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    if (!emailRegex.test(value)) return 'E-mail invalido';
  }

  if (value && field.type === 'phone') {
    const phoneRegex = /^[\d\s()+-]{8,}$/;
    if (!phoneRegex.test(value)) return 'Telefone invalido';
  }

  if (value && field.type === 'select' && field.options) {
    const validValues = field.options.map((o) => o.value);
    if (!validValues.includes(value)) return 'Valor nao e uma opcao valida';
  }

  return null;
}

function validateAll(fields: LeadFormField[]): FieldErrors {
  const errors: FieldErrors = {};
  for (const field of fields) {
    const error = validateField(field, fieldValues[field.key]);
    if (error) {
      errors[field.key] = error;
    }
  }
  return errors;
}

// Submit handler
async function handleSubmit() {
  const errors = validateAll(fields);
  if (Object.keys(errors).length > 0) {
    setFieldErrors(errors);
    return;
  }

  setSubmissionState('loading');
  setSubmitError('');

  try {
    // If embed mode, notify parent we're submitting
    if (isEmbed && window.parent !== window) {
      window.parent.postMessage({ type: 'saaso:form-submitting' }, '*');
    }

    const response = await fetch(`${process.env.NEXT_PUBLIC_API_URL}/forms/${slug}/submit`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ payload: fieldValues }),
    });

    if (!response.ok) {
      const errorData = await response.json().catch(() => ({}));
      const message = errorData.message || errorData.error || 'Erro ao enviar formulario';
      throw new Error(message);
    }

    const result = await response.json();
    setSubmissionState('success');

    if (isEmbed && window.parent !== window) {
      window.parent.postMessage({ type: 'saaso:form-submitted', cardId: result.cardId }, '*');
      // Final height sync after success screen renders
      setTimeout(() => {
        const height = document.documentElement.scrollHeight;
        window.parent.postMessage({ type: 'saaso:form-resize', height }, '*');
      }, 100);
    }
  } catch (error: unknown) {
    setSubmissionState('error');
    setSubmitError(error instanceof Error ? error.message : 'Erro ao enviar formulario');

    if (isEmbed && window.parent !== window) {
      window.parent.postMessage({ type: 'saaso:form-error' }, '*');
    }
  }
}

// CSP meta tag when embed mode
// Add to the page <head> or as a <meta> tag:
// <meta http-equiv="Content-Security-Policy" content="default-src 'self'; script-src 'self' 'unsafe-inline'; style-src 'self' 'unsafe-inline'; frame-ancestors *;">
```

The `sandbox` attribute restricts the iframe:
- `allow-scripts`: enables postMessage and JS execution
- `allow-same-origin`: allows the form to access its own origin (needed for form submission)
- `allow-forms`: allows form submission within the iframe
- No `allow-popups` — prevents target="_blank" abuse
- No `allow-top-navigation` — prevents iframe from redirecting parent

```
postMessage Protocol Documentation:
====================================

Events emitted from the embedded form iframe to the parent window:

1. saaso:form-resize
   { type: 'saaso:form-resize', height: number }
   Sent when form content height changes (dynamic fields, validation errors).
   Parent should update iframe height: iframe.style.height = data.height + 'px'

2. saaso:form-submitting
   { type: 'saaso:form-submitting' }
   Sent when user clicks submit and form is being sent to the backend.
   Parent can show a loading indicator.

3. saaso:form-submitted
   { type: 'saaso:form-submitted', cardId?: string }
   Sent when submission succeeds. cardId is included if a CRM card was created.
   Parent can navigate away, show confirmation, or track conversion.

4. saaso:form-error
   { type: 'saaso:form-error' }
   Sent when submission fails (network error, rate limit, validation error).
   Parent can show error UI or retry logic.

Optional callback registration:
  window.saasoFormCallbacks = {
    onSubmitting: () => { console.log('Form submitting...'); },
    onSubmit: (cardId) => { console.log('Submitted! Card:', cardId); },
    onError: () => { console.log('Submission failed'); },
  };
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Public form page -> POST /forms/:slug/submit | No auth — client-side validation is UX only; server validation is authoritative |
| Embedded iframe -> parent window | postMessage with source validation prevents spoofed events from malicious iframes |
| Parent window -> embedded iframe | No direct communication required; one-way iframe-to-parent events |
| CSP headers in embed mode | Restrict resource loading to prevent XSS in embedded context |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-3-31 | Spoofing | postMessage event from malicious source | mitigate | `event.source === iframe.contentWindow` check in parent script — only trust events from our iframe |
| T-3-32 | Tampering | Client-side validation bypass | mitigate | Server validates all submissions (Plan 03-02); client validation is UX-only |
| T-3-33 | Information Disclosure | CSP headers missing in embed mode | mitigate | Add CSP meta tag when embed=1; production should use Next.js middleware for header-based CSP |
| T-3-34 | Spoofing | iframe sandbox escape | mitigate | sandbox="allow-scripts allow-same-origin allow-forms" — no allow-popups, no allow-top-navigation |
| T-3-35 | Elevation of Privilege | postMessage height inflation (DoS) | accept | Parent caps minimum height at 520px; no maximum cap needed — iframe height is cosmetic |
| T-3-36 | Repudiation | Embedded form submission without tracking | mitigate | Backend creates LeadFormSubmission record; parent-side callbacks can forward events to analytics |
</threat_model>
