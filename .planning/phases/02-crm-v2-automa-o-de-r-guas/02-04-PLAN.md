```typescript
async initiateProactiveIfAssigned(
  cardId: string,
  stageId: string,
  tenantId: string,
): Promise<void> {
  const stage = await this.prisma.stage.findFirst({
    where: { id: stageId, pipeline: { tenantId } },
    include: { agents: true },
  });
  if (!stage || stage.agents.length === 0) return;

  const agent = stage.agents[0];
  const card = await this.prisma.card.findFirst({
    where: { id: cardId, tenantId },
    include: { contact: true },
  });
  if (!card?.contact?.phone) return;

  // Build proactive greeting prompt using existing AgentPromptBuilder
  // Inject classificationCriteria into prompt context
  const prompt = await this.promptBuilder.buildProactivePrompt(agent, card, stage.classificationCriteria);
  const greeting = await this.llmService.generate(prompt); // use existing llm integration

  // Send via WhatsApp
  await this.whatsappService.sendMessage({
    tenantId,
    to: card.contact.phone,
    body: greeting,
  });

  // Create conversation + activity
  await this.prisma.agentConversation.create({
    data: { tenantId, cardId, agentId: agent.id, contactId: card.contactId!, status: 'OPEN' },
  });
  await this.prisma.cardActivity.create({
    data: {
      cardId,
      type: 'AGENT_PROACTIVE',
      content: `Agente ${agent.name} iniciou conversa (D0)`,
      actorId: null,
    },
  });
}

async toggleTakeover(conversationId: string, tenantId: string, userId?: string): Promise<{ status: string }> {
  const conv = await this.prisma.agentConversation.findFirst({ where: { id: conversationId, tenantId } });
  if (!conv) throw new NotFoundException('Conversation not found');
  const newStatus = conv.status === 'OPEN' ? 'HANDOFF_REQUIRED' : 'OPEN';
  await this.prisma.agentConversation.update({ where: { id: conversationId }, data: { status: newStatus } });
  await this.prisma.cardActivity.create({
    data: {
      cardId: conv.cardId,
      type: newStatus === 'HANDOFF_REQUIRED' ? 'AGENT_HANDOFF_MANUAL' : 'AGENT_RESUMED',
      content: newStatus === 'HANDOFF_REQUIRED' ? 'SDR assumiu conversa' : 'Conversa devolvida ao agente',
      actorId: userId,
    },
  });
  return { status: newStatus };
}
```

```typescript
async setStageAgent(
  tenantId: string,
  stageId: string,
  dto: { agentId: string | null; classificationCriteria?: string | null },
) {
  const stage = await this.prisma.stage.findFirst({ where: { id: stageId, pipeline: { tenantId } } });
  if (!stage) throw new NotFoundException('Stage not found');
  if (dto.agentId) {
    const agent = await this.prisma.agent.findFirst({ where: { id: dto.agentId, tenantId } });
    if (!agent) throw new NotFoundException('Agent not found');
  }
  return this.prisma.$transaction([
    // Clear previous assignment (since Agent.stageId is FK to Stage) — any agents with this stageId who are not the new one get stageId=null
    this.prisma.agent.updateMany({ where: { tenantId, stageId }, data: { stageId: null } }),
    ...(dto.agentId ? [this.prisma.agent.update({ where: { id: dto.agentId }, data: { stageId } })] : []),
    this.prisma.stage.update({ where: { id: stageId }, data: { classificationCriteria: dto.classificationCriteria ?? null } }),
  ]);
}
```
